\pgfplotstableset{
  columns/task1/.style = {
      column name = {Sub-task A},
      column type = {l},
      string type
    },
  columns/task2/.style = {
      column name = {Sub-task B},
      column type = {l},
      string type
    },
  columns/experiment1/.style = {
      column name = {Model A},
      column type = {l},
      string type
    },
  columns/experiment2/.style = {
      column name = {Model B},
      column type = {l},
      string type
    },
  columns/tstatistic/.style = {
      column name = {$t$-statistic},
      column type = {r},
      precision = 2
    },
  columns/pvalue/.style = {
      column name = {$p$-value},
      column type = {r},
      precision = 2,
      sci,
      sci zerofill,
    },
  columns/significant/.style = {
      column name = {$p$ < 0.05},
      assign cell content/.code = {
          \ifnum\pdfstrcmp{##1}{True}=0
            \pgfkeyssetvalue{/pgfplots/table/@cell content}{\checkmark}
          \else
            \pgfkeyssetvalue{/pgfplots/table/@cell content}{}
          \fi
        },
    },
}
